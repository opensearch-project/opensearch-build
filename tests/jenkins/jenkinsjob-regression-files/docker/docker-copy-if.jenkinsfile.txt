   docker-copy-lf.run()
      docker-copy-lf.modernSCM({$class=GitSCMSource, remote=https://github.com/opensearch-project/opensearch-build-libraries.git})
      docker-copy-lf.library({identifier=jenkins@13.2.0, retriever=null})
      docker-copy-lf.pipeline(groovy.lang.Closure)
         docker-copy-lf.timeout({time=30})
         docker-copy-lf.echo(Executing on agent [label:none])
         docker-copy-lf.stage(Parameters Check, groovy.lang.Closure)
            docker-copy-lf.script(groovy.lang.Closure)
         docker-copy-lf.stage(Copy Image to ECR/DockerHub, groovy.lang.Closure)
            docker-copy-lf.script(groovy.lang.Closure)
               docker-copy-lf.copyContainer({allTags=false, sourceImage=ci-runner:testtag, sourceRegistry=opensearchstaging, destinationImage=ci-runner:testtag2, destinationRegistry=public.ecr.aws/opensearchstaging})
                  copyContainer.echo(Login Docker Staging)
                  copyContainer.withSecrets({secrets=[{envVar=DOCKER_USERNAME, secretRef=op://opensearch-release-secrets/dockerhub-staging-credentials/username}, {envVar=DOCKER_PASSWORD, secretRef=op://opensearch-release-secrets/dockerhub-staging-credentials/password}]}, groovy.lang.Closure)
                     copyContainer.sh(set +x && echo DOCKER_PASSWORD | docker login --username DOCKER_USERNAME --password-stdin)
                  copyContainer.echo(Copying single image tag from opensearchstaging/ci-runner:testtag to public.ecr.aws/opensearchstaging/ci-runner:testtag2)
                  copyContainer.sh(set -x && crane cp opensearchstaging/ci-runner:testtag public.ecr.aws/opensearchstaging/ci-runner:testtag2)
                  copyContainer.sh(set +x && docker logout && docker logout public.ecr.aws)
         docker-copy-lf.script(groovy.lang.Closure)
            docker-copy-lf.postCleanup()
               postCleanup.cleanWs({disableDeferredWipeout=true, deleteDirs=true})
            docker-copy-lf.sh(docker logout && docker logout public.ecr.aws && docker image prune -f --all)
